### 安装

```
npm i @hhm1999/x-components --save
```

### 引入 x-components

```
import Vue from 'vue';
import xComponents from '@hhm1999/x-components';
import App from './App.vue';

Vue.use(xComponents);

new Vue({
  el: '#app',
  render: h => h(App)
});
```
